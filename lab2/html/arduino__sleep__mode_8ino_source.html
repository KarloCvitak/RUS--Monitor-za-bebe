159 void setupWatchdog(int timerPrescaler) {
160 // Resetiranje watchdog timera
161 wdt_reset();
162
163 // Postavljanje WDT registara za omogućavanje prekida
164 byte wdtBits = timerPrescaler & 7;
165 if (timerPrescaler > 7) {
166 wdtBits |= (1 << 5); // Postavi WDP3 bit
167 }
168
169 // Posebna sekvenca za promjenu registara
170 MCUSR &= ~(1 << WDRF); // Očisti WDRF flag
171 WDTCSR |= (1 << WDCE) | (1 << WDE); // Omogući promjenu
172 WDTCSR = (1 << WDIE) | wdtBits; // Postavi novi timeout i WDT interrupt enable bit
173 }

181 void enterSleep() {
182 // Isključi periferije koje se ne koriste u sleep modu
183 power_adc_disable();

187 // Samo u IDLE modu, Timer0 i USART mogu ostati aktivni
188 if (sleepMode != SLEEP_MODE_IDLE) {
189 power_timer0_disable();
190 power_usart0_disable();
191 }
192
193 set_sleep_mode(sleepMode); // Postavi odabrani sleep mode
194
195 sleep_enable(); // Omogući sleep mode
196
197 // Samo kod PWR_DOWN i STANDBY modova, omogući BOD disable za minimalnu potrošnju
198 if (sleepMode == SLEEP_MODE_PWR_DOWN || sleepMode == SLEEP_MODE_STANDBY) {
199 // Posebna sekvenca za isključivanje BOD-a tijekom sleep moda (samo za AVR s BOD disable bit-om)
200 #if defined(BODS) && defined(BODSE)
201 sleep_bod_disable();
202 #endif
203 }

205 // Ulazak u sleep mode
206 sleep_mode();

209 sleep_disable(); // Onemogući sleep odmah nakon buđenja

211 // Ponovno omogući potrebne periferije
212 power_all_enable(); // Uključi sve periferije

221 void changeMode() {
222 // Ciklički izmijeni sleep modove
223 switch (sleepMode) {
224 case SLEEP_MODE_PWR_DOWN:
225 sleepMode = SLEEP_MODE_PWR_SAVE;
226 Serial.println(F("Novi mod: SLEEP_MODE_PWR_SAVE"));
227 break;
228 case SLEEP_MODE_PWR_SAVE:
229 sleepMode = SLEEP_MODE_STANDBY;
230 Serial.println(F("Novi mod: SLEEP_MODE_STANDBY"));
231 break;
232 case SLEEP_MODE_STANDBY:
233 sleepMode = SLEEP_MODE_IDLE;
234 Serial.println(F("Novi mod: SLEEP_MODE_IDLE"));
235 break;
236 case SLEEP_MODE_IDLE:
237 sleepMode = SLEEP_MODE_PWR_DOWN;
238 Serial.println(F("Novi mod: SLEEP_MODE_PWR_DOWN"));
239 break;
240 default:
241 sleepMode = SLEEP_MODE_PWR_DOWN;
242 Serial.println(F("Reset na: SLEEP_MODE_PWR_DOWN"));
243 break;
244 }
245
246 Serial.println(F("Usporedba potrošnje energije:"));
247 Serial.println(F("PWR_DOWN: Najniža potrošnja (~0.1μA), sve periferije isključene"));
248 Serial.println(F("PWR_SAVE: Niska potrošnja (~1μA), samo Timer2 aktivan"));
249 Serial.println(F("STANDBY: Slično PWR_DOWN, ali glavni oscilator radi (~0.8μA)"));
250 Serial.println(F("IDLE: Viša potrošnja (~4-15mA), CPU zaustavljen, većina periferija aktivna"));
251 }
