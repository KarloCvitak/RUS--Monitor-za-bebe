239 void setupWatchdog(int timerPrescaler) {
240 // Resetiranje watchdog timera
241 wdt_reset();
242
243 // Postavljanje WDT registara za omogućavanje prekida
244 byte wdtBits = timerPrescaler & 7;
245 if (timerPrescaler > 7) {
246 wdtBits |= (1 << 5); // Postavi WDP3 bit
247 }
248
249 // Posebna sekvenca za promjenu registara
250 MCUSR &= ~(1 << WDRF); // Očisti WDRF flag
251 WDTCSR |= (1 << WDCE) | (1 << WDE); // Omogući promjenu
252 WDTCSR = (1 << WDIE) | wdtBits; // Postavi novi timeout i WDT interrupt enable bit
253 }

262 void enterSleep() {
263 // Isključi periferije koje se ne koriste u sleep modu
264 power_adc_disable();
265 power_spi_disable();
266 power_twi_disable();
267
268 // Samo u IDLE modu, Timer0 i USART mogu ostati aktivni
269 if (sleepMode != SLEEP_MODE_IDLE) {
270 power_timer0_disable();
271 power_usart0_disable();
272 }
273
274 noInterrupts(); // Onemogući prekide prije ulaska u sleep
275 set_sleep_mode(sleepMode); // Postavi odabrani sleep mode
276 sleep_enable(); // Omogući sleep mode
277
278 // Samo kod PWR_DOWN i STANDBY modova, omogući BOD disable za minimalnu potrošnju
279 if (sleepMode == SLEEP_MODE_PWR_DOWN || sleepMode == SLEEP_MODE_STANDBY) {
280 #if defined(BODS) && defined(BODSE)
281 sleep_bod_disable();
282 #endif
283 }
284
285 interrupts(); // Ponovno omogući prekide
286 sleep_cpu(); // Ulazak u sleep mode
287
288 // Program nastavlja ovdje nakon prvog buđenja
289 sleep_disable(); // Odmah onemogući sleep
290
291 // Ostani u petlji dok se ne dogodi prekid
292 while (isSleeping) {}
293
294 // Ponovno omogući potrebne periferije
295 power_all_enable(); // Uključi sve periferije
296 }
