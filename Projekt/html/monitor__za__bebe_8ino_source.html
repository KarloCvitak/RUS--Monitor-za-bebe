376 void enterLightSleep(uint64_t sleepDuration) {
377 Serial.println("Ulazak u light sleep mod uz očuvanje WiFi veze...");
378 Serial.println("Broj normalnih očitanja prije sleep-a: " + String(normalReadingsCount));
379 Serial.flush();
380
381 // Spremamo trenutnu vrijednost vremena rada prije spavanja

384 // Postavke za light sleep - očuvanje WiFi veze, ali bez automatskog prekida veze
385 WiFi.setSleep(false); // Postaviti WiFi u aktivan način
386 esp_wifi_set_ps(WIFI_PS_MIN_MODEM);

388 // Isključi vanjsko buđenje ako idemo u normalni sleep
389 if (sleepDuration == SLEEP_DURATION_NORMAL) {
390 // Za dulji sleep, oslanjamo se samo na timer
391 // Ne želimo da zvuk prekida duži sleep
392 esp_sleep_disable_wakeup_source(ESP_SLEEP_WAKEUP_EXT0);

395 esp_sleep_enable_ext0_wakeup((gpio_num_t)SOUND_SENSOR_PIN, HIGH);
396 }

402 esp_light_sleep_start();
403
404 // Kod se nastavlja odavde nakon buđenja iz light sleep-a
405 esp_sleep_wakeup_cause_t wakeup_reason = esp_sleep_get_wakeup_cause();
406 wakeupReason = (uint8_t)wakeup_reason; // Spremi razlog buđenja za debagiranje
407
408 Serial.println("Probuđen iz light sleep moda, razlog: " + getWakeupReasonString(wakeup_reason));
409
410 // Ako je buđenje zbog zvuka, resetiramo brojač normalnih očitanja
411 if (wakeup_reason == ESP_SLEEP_WAKEUP_EXT0) {
412 Serial.println("Buđenje zbog zvuka, resetiranje brojača normalnih očitanja");
413 normalReadingsCount = 0;
