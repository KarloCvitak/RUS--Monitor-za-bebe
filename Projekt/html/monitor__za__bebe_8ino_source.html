256void loop() {
257 // Indikacija da je sustav aktivan

260 bool anyAlert = false;
261
262 // Provjera senzora
263 if (readDHTSensor()) {
264 bool tempAlert = checkTemperature();
265 bool humAlert = checkHumidity();
266 anyAlert = tempAlert || humAlert;
267 }
268
269 bool cryingAlert = checkSound();
270 anyAlert = anyAlert || cryingAlert;
271
272 // Slanje periodičkog izvještaja
273 sendStatusReport();

275 // Isključivanje LED indikatora
276 digitalWrite(STATUS_LED_PIN, LOW);
277
278 // Odlučivanje o ulasku u sleep mod
279 if (!anyAlert) {
280 normalReadingsCount++;

283 // Dovoljno normalnih očitanja, ulazak u duži sleep
284 normalReadingsCount = 0;
285 enterDeepSleep(SLEEP_DURATION_NORMAL);
286 } else {
287 // Još uvijek pratimo da vidimo je li stanje stabilno
288 Serial.println("Normalno čitanje #" + String(normalReadingsCount) + " od " + String(MAX_NORMAL_READINGS));
289 delay(2000);
290 }
291 } else {
292 // Resetiranje brojača normalnih očitanja
293 normalReadingsCount = 0;
294
