138/*
139 * Funkcija za inicijalizaciju Timer1
140 * Konfigurira Timer1 da generira prekid svake 1 sekunde
141 * Koristi CTC (Clear Timer on Compare Match) mode
142 */
143void setupTimer1() {
144 cli(); // Isključi prekide tijekom konfiguracije (Critical Section)
145
146 // Resetiranje registara Timer1
147 TCCR1A = 0; // Timer/Counter Control Register A
148 TCCR1B = 0; // Timer/Counter Control Register B
149 TCNT1 = 0; // Timer/Counter početna vrijednost
150
151 // Postavi Compare Match Register (OCR1A) za 1Hz frekvenciju prekida
152 // Formula: OCR1A = (CPU_freq / (prescaler * desired_freq)) - 1
153 // Za 16MHz CPU, prescaler 1024 i 1Hz: OCR1A = (16MHz / (1024 * 1Hz)) - 1 = 15624
154 OCR1A = 15624;
155
156 // Uključi CTC mode (WGM12) - broji do vrijednosti OCR1A i resetira se
157 TCCR1B |= (1 << WGM12);
158
159 // Postavi prescaler na 1024 (CS12 i CS10 bitovi)
160 TCCR1B |= (1 << CS12) | (1 << CS10);
161
162 // Omogući Timer Compare A prekid
163 TIMSK1 |= (1 << OCIE1A);
164
165 sei(); // Ponovno uključi prekide
166}

168/*
169 * Funkcija za čitanje temperature s DHT22 senzora
170 * Vraća true ako je očitanje uspješno, false ako nije
171 */
172bool readDHT22() {
173 // Čitanje temperature i vlažnosti
174 humidity = dht.readHumidity();
175 temperature = dht.readTemperature();
176
177 // Provjera jesu li očitanja valjana (nisu NaN)
178 if (isnan(temperature) || isnan(humidity)) {
179 Serial.println("Greška u čitanju s DHT22 senzora!");
180 return false;
181 }
182
183 return true;
184}

215 // FALLING detekcija - prekid se aktivira kad pin prijeđe iz HIGH u LOW

229/*
230 * Glavna petlja (loop) - kontinuirano izvršavanje programa
231 * Upravlja LED diodama i provjerava temperaturu
232 * Koristi neblokirajuće tehnike s millis() umjesto delay()
233 */
234void loop() {
235 // Dohvati trenutno vrijeme za upravljanje vremenski ovisnim događajima
236 unsigned long current_time = millis();
237
238 // -------- Upravljanje LED_INT0 (BUTTON0) --------
239 if (led_int0_active && (current_time - led_int0_start_time >= LED_BLINK_DURATION)) {
240 led_int0_active = false; // LED više nije aktivna
241 digitalWrite(LED_INT0_PIN, LOW); // Isključi LED
242 Serial.println("LED_INT0 isključena nakon 1s");
243 }
244
245 // -------- Upravljanje LED_INT1 (BUTTON1) --------
246 if (led_int1_active && (current_time - led_int1_start_time >= LED_BLINK_DURATION)) {
247 led_int1_active = false;
248 digitalWrite(LED_INT1_PIN, LOW);
249 Serial.println("LED_INT1 isključena nakon 1s");
250 }
251
252 // -------- Upravljanje LED_INT2 (BUTTON2) --------
253 if (led_int2_active && (current_time - led_int2_start_time >= LED_BLINK_DURATION)) {
254 led_int2_active = false;
255 digitalWrite(LED_INT2_PIN, LOW);
256 Serial.println("LED_INT2 isključena nakon 1s");
257 }
258
259 // -------- Upravljanje LED_Timer (TIMER1) --------
260 if (led_timer_active && (current_time - led_timer_start_time >= TIMER_LED_DURATION)) {
261 led_timer_active = false;
262 digitalWrite(LED_TIMER_PIN, LOW);
263 Serial.println("Timer prekid aktiviran");
264 }
265
266 // -------- Mjerenje temperature s DHT22 i upravljanje LED_Temp --------
267 // Periodično mjerenje temperature (svakih 2000ms) - najmanjeg prioriteta
268 // DHT22 ne bi trebalo očitavati češće od 2 sekunde
269 if (current_time - last_temp_check_time >= TEMP_CHECK_INTERVAL) {
270 last_temp_check_time = current_time;
271
272 // Očitaj temperaturu i vlažnost
273 if (readDHT22()) {
274 // Provjeri je li temperatura ispod praga
275 if (temperature < TEMP_THRESHOLD) {
276 // Temperatura je ispod praga - aktiviraj treptanje LED
277 led_temp_active = true;
278 Serial.print("Niska temperatura detektirana: ");
279 Serial.print(temperature);
280 Serial.print(" °C, Vlažnost: ");
281 Serial.print(humidity);
282 Serial.println(" %");
283 } else {
284 // Temperatura je iznad praga - isključi LED
285 led_temp_active = false;
286 digitalWrite(LED_TEMP_PIN, LOW);
287
288 Serial.print("Temperatura: ");
289 Serial.print(temperature);
290 Serial.print(" °C, Vlažnost: ");
291 Serial.print(humidity);
292 Serial.println(" %");
293 }
294 }
295 }
296
297 // -------- Upravljanje treptanjem LED_Temp --------
298 // Ako je temperatura niska, LED treperi svakih 200ms
299 if (led_temp_active) {
300 if (current_time - last_temp_toggle_time >= TEMP_BLINK_INTERVAL) {
301 last_temp_toggle_time = current_time;
302 // Toggle LED stanje (ON->OFF ili OFF->ON)
303 digitalWrite(LED_TEMP_PIN, !digitalRead(LED_TEMP_PIN));
304 }
305 }
306}
